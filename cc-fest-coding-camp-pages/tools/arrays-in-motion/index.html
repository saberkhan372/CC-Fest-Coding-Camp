Moving Sentence Machine

A starter sketch that turns generated phrases into moving text objects, combining arrays, language, and motion.

Click to generate new sentences, then press any key to pause or resume the motion.

Try / Notice / Remix

Try this

Notice this

The sketch stores whole sentence objects with both text and position data, not just single words.

Remix this

Make the sentences fall, bounce, fade, or react to the mouse instead of only sliding sideways.

Starter Code

A nice seed for arrays of objects and simple interaction state.

let sentences = [];
let xspeed = 2;
let state = -1;

let verbs = ["runs", "jumps", "swims"];
let nouns = ["saber", "meghna", "anaya"];
let adverbs = ["fast", "slowly", "quietly"];

function setup() {
  createCanvas(400, 400);
  textSize(20);
}

function draw() {
  background(255);
  for (let s of sentences) {
    text(s.text, s.x, s.y);
    s.x += xspeed;
  }

  xspeed = state === -1 ? 0 : 2;
}

function mousePressed() {
  sentences = [];
  for (let i = 0; i < 3; i++) {
    let phrase = random(nouns) + " " + random(verbs) + " " + random(adverbs);
    sentences.push({ text: phrase, x: 0, y: i * 100 + 50 });
  }
}

function keyPressed() {
  state = state * -1;
}
